### Database

  id TEXT PRIMARY KEY,
  profile_id TEXT NOT NULL REFERENCES profiles(id),
  instagram_username TEXT NOT NULL,
  followers_count INTEGER,
  top_posts_data TEXT,
  last_updated DATETIME,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
